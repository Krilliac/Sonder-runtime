# Training-only dependencies for qlora_train.py.
# NOT installed by default — this repo's normal test/runtime deps (requirements-dev.txt)
# stay light. Install these only when you're ready to actually run a training pass:
#     ./venv/Scripts/python.exe -m pip install -r requirements-train.txt
#
# See TRAINING.md for full setup + feasibility notes before installing/running.

# --- PyTorch -----------------------------------------------------------------
# Install a CUDA build matching your driver, NOT the plain PyPI CPU wheel.
# On Windows, the CUDA wheels are hosted on a separate index, e.g. for CUDA 12.1:
#   pip install torch --index-url https://download.pytorch.org/whl/cu121
# Check `nvidia-smi` for your driver's supported CUDA version first.
# Security floor: Dependabot advisories affect older PyTorch ranges through
# 2.12.x, so keep this on a current training-only line.
torch>=2.13,<2.14

# --- Hugging Face stack --------------------------------------------------------
# Security floor: 5.3.0 includes fixes for the ReDoS/RCE advisories reported
# against the earlier 4.x training range.
transformers>=5.3,<6
# Transformers 5.x enforces this at TrainingArguments construction time.
accelerate>=1.1,<2
peft>=0.11,<0.14
datasets>=2.19,<2.21

# --- 4-bit quantization ---------------------------------------------------------
# Keep this aligned with the CUDA 13-capable native Windows environment used by
# the attended training path. Older 0.43/0.44 wheels do not match that runtime.
# CPU-offload paths remain disabled by the lifecycle controller. If a supported
# wheel is unavailable for a future Python/CUDA pair, the smoother path is WSL2:
#   wsl --install   (if not already set up)
#   then re-create the venv and pip install inside the WSL2 Linux environment,
#   where bitsandbytes' official Linux wheels apply cleanly.
bitsandbytes>=0.49,<0.50
